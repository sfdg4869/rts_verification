-- Case1 에서 사용할 프로시저 생성 (한번 수행될 때 3초 동안 도는 프로시저)
CREATE OR REPLACE PROCEDURE qs_sql_test_proc1 as 
    v_start TIMESTAMP; 
BEGIN
    v_start := SYSTIMESTAMP;
           
    -- qs_sql_test_proc1 Elapsed Time 3 seconds
    WHILE (SYSTIMESTAMP - v_start) < INTERVAL '3' SECOND LOOP
            NULL;
    END LOOP;
END;
